find_package(rviz_rendering REQUIRED)

# so that they can be found at runtime. Requires rviz_rendering.

  )

  rclcpp
  rclpy
  geometry_msgs
  std_msgs
  std_srvs
)

  find_package(${dependency} REQUIRED)
endforeach()
